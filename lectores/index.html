// Worker: lectores-api (v2) // // Endpoints: // GET /state -> { signups: n, reactions: [ {id, handle, ig_link, category, chapter, phrase, created_at, replies:[...]}, ... ] } // POST /signup -> { ok: true } // POST /react -> { ok: true, id: } (acepta reply_to para respuestas anidadas) // // Bindings: D1 "DB", KV "RL" (opcional para rate limiting). const CORS = { "access-control-allow-origin": "*", "access-control-allow-methods": "GET, POST, OPTIONS", "access-control-allow-headers": "content-type", "access-control-max-age": "86400", }; function json(body, status = 200, extra = {}) { return new Response(JSON.stringify(body), { status, headers: { "content-type": "application/json; charset=utf-8", ...CORS, ...extra }, }); } const BANNED = [ // spam / SEO "http://", "https://", "www.", ".com", ".net", ".org", // insultos obvios (objetivo: romper el auto-piloto, no censurar opinión) "mierda", "puta", "gilipollas", "cabron", "cabrón", "coño", "joder", "pendejo", // auto-promo "sígueme", "siganme", "follow me", "mi perfil", ]; function looksLikeSpam(s) { if (!s) return false; const t = s.toLowerCase(); for (const w of BANNED) if (t.includes(w)) return true; if (/(.)\1{5,}/.test(t)) return true; const letters = s.replace(/[^A-Za-zÁÉÍÓÚÜÑáéíóúüñ]/g, ""); if (letters.length > 20) { const upper = letters.replace(/[^A-ZÁÉÍÓÚÜÑ]/g, "").length; if (upper / letters.length > 0.7) return true; } return false; } // Handle válido: si viene marcado como IG, solo letras, números, . y _ (reglas de IG). // Si es alias libre, permitimos más (espacios, tildes, etc.) pero acotamos longitud. function sanitizeHandle(h, isIgLink) { if (!h) return ""; let out = String(h).trim().replace(/^@/, ""); if (isIgLink) { out = out.replace(/[^a-zA-Z0-9._]/g, "").slice(0, 30); } else { // Alias libre: quitar caracteres de control y limitar out = out.replace(/[<>"'\u0000-\u001f]/g, "").slice(0, 40); } return out; } function validIgHandle(h) { return /^[a-zA-Z0-9._]{2,30}$/.test(h); } function validEmail(e) { return /^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(e) && e.length < 200; } async function rateLimit(env, key, windowSec = 60, max = 3) { if (!env.RL) return true; const k = "rl:" + key; const cur = parseInt((await env.RL.get(k)) || "0", 10); if (cur >= max) return false; await env.RL.put(k, String(cur + 1), { expirationTtl: windowSec }); return true; } async function handleState(env) { const sRow = await env.DB.prepare("SELECT COUNT(*) AS c FROM signups").first(); // Traemos todas las reacciones recientes (roots + replies) y luego agrupamos. const all = await env.DB.prepare( `SELECT id, handle, ig_link, category, chapter, phrase, reply_to, created_at FROM reactions ORDER BY created_at DESC LIMIT 500` ).all(); const rows = all.results || []; const roots = []; const repliesByParent = new Map(); for (const r of rows) { const item = { id: "db-" + r.id, handle: r.handle, ig_link: !!r.ig_link, category: r.category, chapter: r.chapter || "", phrase: r.phrase, created_at: r.created_at, }; if (r.reply_to) { const pid = "db-" + r.reply_to; if (!repliesByParent.has(pid)) repliesByParent.set(pid, []); repliesByParent.get(pid).push(item); } else { roots.push(item); } } // Pegar réplicas en cada root (orden cronológico ascendente dentro del hilo) for (const root of roots) { const reps = repliesByParent.get(root.id) || []; reps.sort((a, b) => a.created_at.localeCompare(b.created_at)); root.replies = reps; } return json({ signups: sRow?.c ?? 0, reactions: roots }); } async function handleSignup(req, env) { const ip = req.headers.get("cf-connecting-ip") || "0"; // Rate limit: máximo 2 signups por IP cada hora (era 5). if (!(await rateLimit(env, "su:" + ip, 3600, 2))) { return json({ error: "Demasiados intentos. Vuelva en un rato." }, 429); } let b; try { b = await req.json(); } catch { return json({ error: "JSON inválido" }, 400); } // Honeypot: si el campo invisible viene relleno, es un bot. Fingimos éxito // para no darle pistas, pero no guardamos nada. if (b.website && String(b.website).trim().length > 0) { return json({ ok: true }); } const email = (b.email || "").trim().toLowerCase(); if (!validEmail(email)) return json({ error: "Email no válido" }, 400); try { await env.DB.prepare( "INSERT INTO signups (email, ip, created_at) VALUES (?1, ?2, ?3)" ).bind(email, ip, new Date().toISOString()).run(); } catch (e) { if (String(e.message || "").includes("UNIQUE")) { return json({ ok: true, note: "ya_apuntado" }); } return json({ error: "No se ha podido registrar" }, 500); } return json({ ok: true }); } async function handleReact(req, env) { const ip = req.headers.get("cf-connecting-ip") || "0"; if (!(await rateLimit(env, "rx:" + ip, 3600, 5))) { return json({ error: "Demasiadas publicaciones desde esta conexión." }, 429); } let b; try { b = await req.json(); } catch { return json({ error: "JSON inválido" }, 400); } const igLink = !!b.ig_link; const handle = sanitizeHandle(b.handle || "", igLink) || "anónimo"; const category = (b.category || "").trim().toUpperCase(); const chapter = (b.chapter || "").trim().slice(0, 80); const phrase = (b.phrase || "").trim().replace(/\s+/g, " "); // Si marca enlace a IG, el handle DEBE ser un IG válido. if (igLink && !validIgHandle(handle)) { return json({ error: "Si desea enlazar Instagram, el @ debe ser válido." }, 400); } if (!["A","B","C","D","E","F"].includes(category)) { return json({ error: "Categoría no válida" }, 400); } if (phrase.length < 5 || phrase.length > 140) { return json({ error: "La frase debe tener entre 5 y 140 caracteres" }, 400); } if (looksLikeSpam(phrase) || looksLikeSpam(handle)) { return json({ error: "Contenido no permitido." }, 400); } // Resolver reply_to: el cliente envía "db-" (ids del servidor) o "seed-N" (semilla local). // Solo aceptamos "db-"; los seed-N no pueden tener respuestas persistidas (el cliente las guarda en memoria). let replyTo = null; if (b.reply_to) { const m = String(b.reply_to).match(/^db-(\d+)$/); if (m) { const parentId = parseInt(m[1], 10); const parent = await env.DB.prepare("SELECT id FROM reactions WHERE id = ?1 AND reply_to IS NULL").bind(parentId).first(); if (!parent) return json({ error: "La entrada a la que responde no existe." }, 400); replyTo = parentId; } else { return json({ error: "No se puede responder a entradas locales. Recargue la página." }, 400); } } // Anti-duplicado simple: misma frase del mismo IP en la última hora. const hourAgo = new Date(Date.now() - 3600_000).toISOString(); const dup = await env.DB.prepare( "SELECT 1 FROM reactions WHERE ip = ?1 AND phrase = ?2 AND created_at > ?3 LIMIT 1" ).bind(ip, phrase, hourAgo).first(); if (dup) return json({ error: "Ya publicó esa misma frase recientemente." }, 409); const res = await env.DB.prepare( "INSERT INTO reactions (handle, ig_link, category, chapter, phrase, reply_to, ip, created_at) VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8)" ).bind( handle, igLink ? 1 : 0, category, chapter || null, phrase, replyTo, ip, new Date().toISOString() ).run(); const newId = res.meta?.last_row_id; return json({ ok: true, id: newId ? "db-" + newId : null }); } export default { async fetch(req, env) { const url = new URL(req.url); if (req.method === "OPTIONS") return new Response(null, { headers: CORS }); try { if (url.pathname === "/state" && req.method === "GET") return handleState(env); if (url.pathname === "/signup" && req.method === "POST") return handleSignup(req, env); if (url.pathname === "/react" && req.method === "POST") return handleReact(req, env); return json({ error: "Not found" }, 404); } catch (e) { return json({ error: "Server error" }, 500); } }, };
